PARTE 4 — BLOCO 2: SQL pós-pipeline
================================================================

  -- 2A: 3 specs no banco
  SELECT id, titulo, status, pipeline_classificacao, score
  FROM knowledge_specs ORDER BY created_at DESC LIMIT 5;

  -- 2B: 6 entradas de auditoria (2 por spec)
  SELECT spec_id, acao, created_at FROM knowledge_spec_audit ORDER BY created_at DESC LIMIT 10;

  -- 2C: knowledge_base ainda vazio
  SELECT COUNT(*) FROM knowledge_base;

  -- 2D: referências presentes (>= 2 por spec)
  SELECT id, titulo, jsonb_array_length(referencias_oficiais) as refs
  FROM knowledge_specs ORDER BY created_at DESC LIMIT 3;

  -- 2E: hash preenchido
  SELECT id, LEFT(hash,16) as hash_prefix FROM knowledge_specs ORDER BY created_at DESC LIMIT 3;

CRITÉRIO:
  3 specs presentes, 6 audits, knowledge_base COUNT=0,
  refs >= 2 em cada spec, hash não-nulo.
